## 1. 测试策略总览

```
                    ┌─────────────┐
                    │   E2E 测试   │  (少量，覆盖核心用户场景)
                    │   ~5%        │
                    └──────┬──────┘
                           │
                    ┌──────┴──────┐

                    │   ~25%       │

// packages/core/src/tools/__tests__/edit.test.ts

import { describe, it, expect, beforeEach } from "vitest";
import { EditTool } from "../builtin/edit.js";
import { createMockContext } from "../../__tests__/helpers.js";

describe("EditTool", () => {
    let tool: EditTool;
    let context: ToolContext;

    beforeEach(() => {
        tool = new EditTool();
        context = createMockContext({
            files: {
                "/project/src/app.ts": 'const x = 1;\nconst y = 2;\n',
            },
        });
    });

    // ─── 正向用例 ───
    it("应正确替换文件中唯一的匹配字符串", async () => {
        const result = await tool.execute({
            file_path: "/project/src/app.ts",
            old_string: "const x = 1;",
            new_string: "const x = 10;",
        }, context);

        expect(result.is_error).toBe(false);
        expect(result.content).toContain("替换了 1 处");
    });

    // ─── 边界条件 ───
    it("当 old_string 在文件中不唯一且未设置 replace_all 时应报错", async () => {
        const result = await tool.execute({
            file_path: "/project/src/app.ts",
            old_string: "const",
            new_string: "let",
        }, context);

        expect(result.is_error).toBe(true);
        expect(result.content).toContain("不唯一");
    });

    // ─── 安全测试 ───
    it("应拒绝工作目录外的路径", async () => {
        const result = await tool.execute({
            file_path: "/etc/passwd",
            old_string: "x",
            new_string: "y",
        }, context);

        expect(result.is_error).toBe(true);
        expect(result.content).toContain("工作目录外");
    });

    // ─── replace_all 模式 ───
    it("replace_all=true 时应替换所有匹配项", async () => {
        const result = await tool.execute({
            file_path: "/project/src/app.ts",
            old_string: "const",
            new_string: "let",
            replace_all: true,
        }, context);

        expect(result.is_error).toBe(false);
        expect(result.content).toContain("替换了 2 处");
    });
});
```

### 3.1 Agent Loop 集成测试（Mock LLM）

```typescript
// packages/core/src/agent/__tests__/loop.integration.test.ts

describe("AgentLoop Integration", () => {
    /** 创建一个模拟 LLM Provider，返回预设的 chunk 序列 */
    function createMockProvider(chunks: ResponseChunk[]): LLMProvider {
        return {
            async *chat() {
                for (const chunk of chunks) {
                    yield chunk;
                }
            },
            contextWindow: () => 200000,
            countTokens: async () => 100,
            supportsFeature: () => true,
        };
    }

    it("应完成 思考→工具调用→观察 的完整循环", async () => {
        const provider = createMockProvider([
            { type: "text", content: "让我查看文件" },
            { type: "tool_use", id: "1", name: "read", input: { file_path: "test.ts" } },
            { type: "stop", reason: "end_turn" },
        ]);

        const config: AgentConfig = {
            provider,
            model: "test",
            maxToolRounds: 5,
            maxTokensPerTurn: 1000,
            systemPrompt: "测试",
            tools: [new ReadTool()],
            thinkingEnabled: false,
        };

        const loop = new AgentLoop();
        const events: TurnEvent[] = [];
        for await (const event of loop.run("读取文件", config)) {
            events.push(event);
        }

        // 验证事件序列
        expect(events).toContainEqual({ type: "text", content: "让我查看文件" });
        expect(events).toContainEqual(
            expect.objectContaining({ type: "tool_call", tool: expect.objectContaining({ name: "read" }) })
        );
    });

    it("达到 maxToolRounds 时应强制停止", async () => {
        // 构造一个永远返回工具调用的 provider
        const provider = createMockProvider(
            Array(10).fill([
                { type: "tool_use", id: "1", name: "read", input: { file_path: "test.ts" } },
                { type: "stop", reason: "end_turn" },
            ]).flat()
        );

        const config: AgentConfig = {
            provider,
            model: "test",
            maxToolRounds: 3, // 限制 3 轮
            // ...
        };

        const loop = new AgentLoop();
        const events: TurnEvent[] = [];
        for await (const event of loop.run("读取文件", config)) {
            events.push(event);
        }

        // 应该只执行了 3 轮工具调用
        const toolCalls = events.filter(e => e.type === "tool_call");
        expect(toolCalls.length).toBeLessThanOrEqual(3);
        expect(events).toContainEqual(
            expect.objectContaining({ type: "error" })
        );
    });
});
```

## 6. CI 流水线

```yaml
# .github/workflows/ci.yml

name: CI
on: [push, pull_request]

jobs:
  test:
    runs-on: ubuntu-latest
    strategy:
      matrix:
        node-version: [20, 22]
    steps:
      - uses: actions/checkout@v4
      - uses: pnpm/action-setup@v2
      - uses: actions/setup-node@v4
        with:
          node-version: ${{ matrix.node-version }}
          cache: "pnpm"
      - run: pnpm install
      - run: pnpm typecheck
      - run: pnpm lint
      - run: pnpm test:coverage
      - run: pnpm build

  e2e:
    runs-on: ubuntu-latest
    needs: test
    steps:
      - uses: actions/checkout@v4
      - uses: pnpm/action-setup@v2
      - uses: actions/setup-node@v4
        with:
          node-version: 22
          cache: "pnpm"
      - run: pnpm install
      - run: pnpm build
      - run: pnpm --filter @y-claude-code/cli test:e2e
        env:
          ANTHROPIC_API_KEY: ${{ secrets.ANTHROPIC_API_KEY }}
```

## 8. 测试检查清单

### 新增工具时

- [ ] 正向用例（正常输入 → 预期输出）
- [ ] 边界条件（空输入、超大输入、特殊字符、不存在的文件）

- [ ] 错误恢复（异常时返回 is_error 而非抛出）

- [ ] 流式解析（text / tool_use / thinking / error 各类型 chunk）

- [ ] 重试逻辑
- [ ] Token 计数准确性

- [ ] 单轮文本对话

- [ ] 权限确认流程
- [ ] 停止条件（自然结束、maxRounds、用户中断）

- [ ] Token 超限时压缩
